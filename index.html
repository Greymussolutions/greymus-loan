CORE OPERATIONAL LOAN LOGIC // ========================================== let registeredClients = []; // Listen to data from your 'clients' collection in Firestore function listenToRegisteredClients() { firebase.firestore().collection('clients').orderBy('name') .onSnapshot((snapshot) => { const selectDropdown = document.getElementById('loan-client-select'); if(!selectDropdown) return; // Fail-safe registeredClients = []; selectDropdown.innerHTML = ''; snapshot.forEach((doc) => { const client = doc.data(); client.id = doc.id; registeredClients.push(client); const opt = document.createElement('option'); opt.value = client.id; opt.textContent = `${client.name} (${client.phone})`; selectDropdown.appendChild(opt); }); }); } // Onboard / Pre-Register Client document.getElementById('btn-register-client')?.addEventListener('click', async () => { const name = document.getElementById('reg-client-name').value.trim(); const phone = document.getElementById('reg-client-phone').value.trim(); if (!name || !phone) return alert("Please fill in both Name and Phone number."); try { await firebase.firestore().collection('clients').add({ name: name, phone: phone, created_at: firebase.firestore.FieldValue.serverTimestamp() }); alert("Client profile saved successfully!"); document.getElementById('reg-client-name').value = ''; document.getElementById('reg-client-phone').value = ''; } catch (err) { alert("Error saving client: " + err.message); } }); // Dynamic Real-Time Calculator Math Engine function calculateLoanBreakdown() { const principalInput = document.getElementById('loan-principal'); const weeksInput = document.getElementById('loan-weeks'); if(!principalInput || !weeksInput) return; const principal = parseFloat(principalInput.value) || 0; const weeks = parseInt(weeksInput.value) || 0; if (principal > 0 && weeks > 0) { // Equation logic: (5% * Principal) * Number of weeks const totalInterest = (principal * 0.05) * weeks; const totalRepayable = principal + totalInterest; const weeklyPayment = totalRepayable / weeks; document.getElementById('calc-interest').textContent = `KES ${totalInterest.toLocaleString()}`; document.getElementById('calc-total').textContent = `KES ${totalRepayable.toLocaleString()}`; document.getElementById('calc-weekly').textContent = `KES ${weeklyPayment.toLocaleString()} / week`; } else { document.getElementById('calc-interest').textContent = "KES 0.00"; document.getElementById('calc-total').textContent = "KES 0.00"; document.getElementById('calc-weekly').textContent = "KES 0.00 / week"; } } // Bind live listening input calculations document.getElementById('loan-principal')?.addEventListener('input', calculateLoanBreakdown); document.getElementById('loan-weeks')?.addEventListener('input', calculateLoanBreakdown); // Disburse Loan to Selected Client document.getElementById('btn-issue-loan')?.addEventListener('click', async () => { const clientId = document.getElementById('loan-client-select').value; const principal = parseFloat(document.getElementById('loan-principal').value) || 0; const weeks = parseInt(document.getElementById('loan-weeks').value) || 0; if (!clientId) return alert("Error: You must choose a registered client from the dropdown list first."); if (principal <= 0 || weeks <= 0) return alert("Please enter valid principal values and duration weeks."); const matchedClient = registeredClients.find(c => c.id === clientId); const totalInterest = (principal * 0.05) * weeks; const totalRepayable = principal + totalInterest; const weeklyPayment = totalRepayable / weeks; try { await firebase.firestore().collection('loans').add({ clientId: matchedClient.id, clientName: matchedClient.name, clientPhone: matchedClient.phone, principalAmount: principal, loanWeeksDuration: weeks, calculatedInterest: totalInterest, totalRepayableAmount: totalRepayable, weeklyInstallmentAmount: weeklyPayment, status: "active", issued_at: firebase.firestore.FieldValue.serverTimestamp() }); alert(`Success! Loan approved for ${matchedClient.name}. Weekly collection target: KES ${weeklyPayment.toFixed(2)}`); document.getElementById('loan-client-select').value = ""; document.getElementById('loan-principal').value = ""; document.getElementById('loan-weeks').value = ""; calculateLoanBreakdown(); } catch (err) { alert("Transaction failed: " + err.message); } }); // Run dropdown stream listener on app start listenToRegisteredClients();
